#include <iostream>
#include<GL/gl.h>
#include <GL/glut.h>

void myDisplay(void)
{
glClear (GL_COLOR_BUFFER_BIT);
glColor3f (0.0, 0.0, 1.0);
glPointSize(5.0);

glBegin(GL_LINES);
glVertex2i(0, 240);
glVertex2i(640, 240);
glColor3f (0.0, 0.0, 1.0);
glVertex2i(320, 480);
glVertex2i(320, 0);
glEnd();

glBegin(GL_TRIANGLES); //Triangle 1
glColor3f (0.0, 0.0, 1.0);
glVertex2i(50, 250);
glColor3f (0.0, 0.0, 1.0);
glVertex2i(150, 350);
glColor3f (0.0, 0.0, 1.0);
glVertex2i(250, 250);
glEnd();


glBegin(GL_TRIANGLES); //Triangle 2
glColor3f (0.0, 0.0, 1.0);
glVertex2i(50, 350);
glColor3f (0.0, 0.0, 1.0);
glVertex2i(250, 350);
glColor3f (0.0, 0.0, 1.0);
glVertex2i(150, 450);
glEnd();

glBegin(GL_LINES);     /*X*/
glVertex2i(350, 250);
glVertex2i(610, 450);
glVertex2i(350, 450);
glVertex2i(610, 250);
glVertex2i(350, 450);
glVertex2i(350, 250);
glVertex2i(610, 450);
glVertex2i(610, 250);
glEnd();


glBegin(GL_LINES); //Box
glVertex2i(50, 50);
glVertex2i(250, 210);
glVertex2i(250, 50);
glVertex2i(50, 210);
glVertex2i(50, 210);
glVertex2i(250, 210);
glVertex2i(250, 210);
glVertex2i(250, 50);
glVertex2i(250, 50);
glVertex2i(50, 50);
glVertex2i(50, 50);
glVertex2i(50, 210);
glEnd();



glBegin(GL_LINES);//Square1
glVertex2i(450, 50);
glVertex2i(600, 50);
glVertex2i(600, 150);
glVertex2i(450, 150);
glVertex2i(450, 150);
glVertex2i(450, 50);
glVertex2i(600, 150);
glVertex2i(600, 50);
glEnd();


glBegin(GL_LINES);//Square 2
glVertex2i(400, 100);
glVertex2i(550, 100);
glVertex2i(400, 200);
glVertex2i(550, 200);
glVertex2i(400, 200);
glVertex2i(400, 100);
glVertex2i(550, 200);
glVertex2i(550, 100);



glEnd();



glFlush ();
}

void myInit (void)
{
glClearColor(1.0, 1.0, 1.0, 0.0);
glMatrixMode(GL_PROJECTION);
glLoadIdentity();
gluOrtho2D(0.0, 640.0, 0.0, 480.0);
}

int main(int argc, char** argv)
{
glutInit(&argc, argv);
glutInitDisplayMode (GLUT_SINGLE | GLUT_RGB);
glutInitWindowSize (640, 480);
glutInitWindowPosition (300, 200);
glutCreateWindow ("Assignment 2");
glutDisplayFunc(myDisplay);
myInit ();
glutMainLoop();
}
